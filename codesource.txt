class Player(pygame.sprite.Sprite):
    def __init__(self,personnage, pos_x, pos_y):
        super().__init__()
        
        # liste des images
        self.idle = [
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_idle_anim\knight_m_idle_1.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_idle_anim\knight_m_idle_2.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_idle_anim\knight_m_idle_3.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_idle_anim\knight_m_idle_4.png")
        ]

        self.right = [
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_run_anim\knight_m_run_1.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_run_anim\knight_m_run_2.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_run_anim\knight_m_run_3.png"),
            pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_run_anim\knight_m_run_4.png")
        ]

        self.left = [ pygame.transform.flip(image, 1, 0) for image in self.right ]

        self.hit = [pygame.image.load(f"images\Heroes\Knight\Knight_M\knight_m_hit_anim\knight_m_hit_1.png")]

        self.current_image = 0
        self.image = self.right[self.current_image]
        self.direction = "right"
        self.is_right = True

        # updating
        self.position = [pos_x, pos_y]
        self.old_position = self.position.copy()
        self.rect = self.image.get_rect()
        self.width = self.image.get_width()
        self.height = self.image.get_height()

        self.feet = pygame.Rect(0, 0, self.width, 8)

    def update(self):
        
        self.rect.topleft = self.position
        self.feet.midbottom = self.rect.midbottom

    def animate(self, animation, spd_img):
        self.current_image += spd_img
        if self.current_image >= len(animation):
            self.current_image = 0

        self.image = animation[int(self.current_image)]

    def move_left(self, speed):
        if self.direction != "left":
            self.image = self.left[0]
            self.direction = "left"
            self.is_right = False

        self.position[0] -= speed

    def move_right(self, speed):
        if self.direction != "right":
            self.image = self.right[0]
            self.direction = "right"
            self.is_right = True
            
        self.position[0] += speed

    def move_up(self, speed):
        self.direction = "up"
        self.position[1] -= speed * 3 / 4

    def move_down(self, speed):
        self.direction = "down"
        self.position[1] += speed * 3 / 4

    def save_location(self): self.old_position = self.position.copy()

    def move_back(self):
        self.position = self.old_position
        self.rect.topleft = self.position
        self.feet.midbottom = self.rect.midbottom

    def idle_animation(self):
        if self.is_right:
            self.animate(self.idle, 0.1)
        else:
            self.animate([pygame.transform.flip(image, 1, 0) for image in self.idle], 0.1)

def handle_input(player, speed, img_speed, controls):    
    pressed = pygame.key.get_pressed()

    if pressed[controls[0]] and player.position[1] > 0: # haut
        player.move_up(speed)
        
        if pressed[controls[2]]:
            pass
        elif pressed[controls[3]]:
            pass
        else:
            if player.is_right:        
                animate(player.right,img_speed)
            else:
                animate(player.left,img_speed)

    if pressed[controls[1]] and player.position[1] < SCREEN.height - player.height : # bas
        player.move_down(speed)

        if pressed[controls[2]]:
            pass
        elif pressed[controls[3]]:
            pass
        else:
            if player.is_right:        
                animate(player.right,img_speed)
            else:
                animate(player.left,img_speed)
    
    if pressed[controls[2]] and player.position[0] > 0: # gauche
        player.move_left(speed)

        animate(player.left, img_speed)
    
    if pressed[controls[3]] and player.position[0] < SCREEN.width - player.width: # droite
        player.move_right(speed)

        animate(player.right, img_speed)
    
    if sum([pressed[ctrl] for ctrl in controls]) == 0:
        player.idle_animation()
